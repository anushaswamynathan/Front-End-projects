<script type="text/javascript">       

/*below code is to show the difference of time between appearance and disappearance of the square and also the appearance of random shapes at random parts of the screen */
function getRandomColor() {
    var letters = '0123456789ABCDEF'.split('');
    var color = '#';
    for (var i = 0; i < 6; i++ ) {
        color += letters[Math.floor(Math.random() * 15)];
    }
    return color;
}

var createdtime;
var clickedtime; 
var reactiontime;

function makebox()
{

	
	var x= Math.random();
	x=x*5000;
	setTimeout(function()
{  
	if(Math.random()>0.5)
	{
		document.getElementById("box").style.borderRadius="100px";
	}
	else
	{
		document.getElementById("box").style.borderRadius="0";
	}
	var top=Math.random();
	top=top*300;
	var left=Math.random();
	left=left*500;

	document.getElementById("box").style.top=top+"px";
	document.getElementById("box").style.left=left+"px";
	document.getElementById("box").style.backgroundColor=getRandomColor();
	document.getElementById("box").style.display="block";
	createdtime=Date.now();
}, x);

}
document.getElementById("box").onclick=function()
{
	clickedtime=Date.now();
	reactiontime=(createdtime-clickedtime)/1000;
	document.getElementById("time").innerHTML=reactiontime;
	//alert(reactiontime);
	document.getElementById("box").style.display="none";
	makebox();

}

makebox();
